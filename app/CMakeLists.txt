cmake_minimum_required (VERSION 3.8)

# Add source to this project's executable.
set(APP_SRC
    "main.cpp"
)

add_executable (vk ${APP_SRC})

target_include_directories(vk PUBLIC "include" ${Vulkan_INCLUDE_DIRS})
target_link_libraries(vk ${Vulkan_LIBRARIES} glfw3::glfw3)
add_dependencies(vk Resources)

add_custom_command(TARGET vk POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        ${GLFW3_DLL}
        $<TARGET_FILE_DIR:vk>
)
